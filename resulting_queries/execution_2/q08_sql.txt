SELECT 
    e.first_name, 
    e.last_name, 
    d.dept_name, 
    s.salary
FROM 
    employees e
JOIN 
    dept_emp de ON e.emp_no = de.emp_no
JOIN 
    departments d ON de.dept_no = d.dept_no
JOIN 
    salaries s ON e.emp_no = s.emp_no AND de.from_date <= s.from_date AND de.to_date >= s.from_date
ORDER BY 
    e.first_name, 
    e.last_name;
