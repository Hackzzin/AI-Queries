```
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?dept ?manager
WHERE {
  ?dept a owl:Class ;
        rdfs:subClassOf <http://www.semanticweb.org/usuario/ontologies/2025/6/employees#Department> .
  ?dept <http://www.semanticweb.org/usuario/ontologies/2025/6/employees#hasManager> ?manager .
  ?manager a owl:NamedIndividual ;
           rdfs:type <http://www.semanticweb.org/usuario/ontologies/2025/6/employees#DepartmentManager> .
}
